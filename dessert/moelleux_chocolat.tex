\begin{recipe}[
 preparationtime = 45 minutes,
 bakingtemperature = {\protect\bakingtemperature{
	fanoven=\unit[200]{ºC}
	}
 },
 bakingtime = 40 minutes,
 source = Christelle Paul
]{Moelleux au chocolat à la courgette}
\ingredients{
	\unit[200]{g} &  Chocolat noir \\
	\unit[200]{g} &  Courgette \\
	4 & Oeufs \\
	\unit[2]{c à s} & Farine \\
	\unit[1/2]{c à c} & Baking powder \\
	\unit[1]{pincée} & Sel
}

	\index{Source!Christelle Paul}
	\index{Dessert!Chocolat}
	\index{Dessert!Courgette}
	\index{Ingrédients!Légumes!Courgette}
	\index{Ingrédients!Sucré!Chocolat}


\preparation {
 \step Préchauffer le four à \unit{150}{ºC}.
 \step Faites fondre le chocolat au bain marie.
 \step Hachez très finement la chair courgette ( sans peau et sans pépins ) et cuisez la à la vapeur.
 \step Mélangez le chocolat fondu et la purée de courgette.
 \step Ajoutez les 4 jaunes d'oeufs, la farine, la baking powder et le sel.
 \step Montez les 4 blancs d'oeufs en neige ferme.
 \step Ajoutez les blancs en neige délicatement à la préparation.
 \step Versez la préparation dans un moule tapissé de papier cuisson.
 \step Cuire au four pendant \unit[40]{minutes} à four chaud.
}
\end{recipe}
