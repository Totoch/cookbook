\begin{recipe}[
 bakingtime=45 minutes,
 portion=4-5,
 source=Josiane Cerfontaine,
]{Pâtes provençales Josiane}
\ingredients{
	3 & Poivrons \\
	3 & Boîtes de 400g de tomates \\
	\unit[100]{g} & Feta en petits cubes \\
	1 & Boule de mozarella \\
	1 & Petite boîte de concentré de tomates \\
	2 & Échallottes \\
	1 & Éclat d'ail \\
	\unit[1]{c à s} & Huile d'olive \\
	\unit[2]{c à s} & Herbes de provences \\
	& Poivre noir \\
	& Sel \\
	\unit[500]{g} & Hachis de porc et boeuf \\
	\unit[1]{c à s} & Beurre \\
}
	\index{Source!Josiane Cerfontaine}
	\index{Plat!Pâtes}
	\index{Ingrédients!Légumes!Boîte de tomates}
	\index{Ingrédients!Légumes!Poivrons}
	\index{Ingrédients!Viandes!Hachis de porc}
	\index{Ingrédients!Viandes!Hachis de boeuf}
	\index{Ingrédients!Fromages!Mozarella}
	\index{Ingrédients!Fromages!Feta}

\preparation{
	\step Façonner les boulettes avec le hachis et les faire dorer dans le beurre, fort puis à feu doux.
	\step Faire blondir l'ail et les échalotes dans une poêle huilée.
	\step Ajouter les tomates et les poibrons en morceaux, ajouter le sel, le poivre et les herves de Provence.
	\step Faire étuver.
	\step Intégrer le concentré de tomates et mijoter à feu couvert doux pendant \unit[25]{minutes}.
	\step Disposer les boulettes dans un grand plat pour le four, y répartir les cuves de feta, couvrir de sauce et gratiner de mozarella.
	\step Faire colorer quelques minutes au grill.
}
\end{recipe}
