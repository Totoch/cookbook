\begin{recipe}[
	bakingtime = 20 minutes,
	bakingtemperature = {\protect\bakingtemperature{
		fanoven=\unit[200]{ºC}
		}
	},
	source = Josiane Cerfontaine
]{Lasagne sauce spaghetti Josiane}
\ingredients{
	& Gruyère rapé \\
	& Feuilles de lasagne \\
	Sauce: & Voir \hyperref[SauceSpaghettiJosiane]{Sauce Spaghetti Josiane} \\
	Béchamelle: & \\
	\unit[40]{g} & Beurre \\
	\unit[40]{g} & Farine \\
	\unit[1/2]{l} & Lait \\
	& Sel \\
	& Poivre \\
	& Noix de muscade \\
}

	\index{Source!Josiane Cerfontaine}
	\index{Plat!Pâtes!Lasagne}
	\index{Plat!Pâtes!Josiane}

\preparation{
	\step Déposer un peu de sauce tomate dans le fond d'un plat.
	\step Déposer un peu de sauce béchamelle par-dessus.
	\step Disposer les feuilles de lasagne sur toute la surface (recouper pour adapter).
	\step Faire 4 étages en terminant par un peu de sauce, un peu de béchamelle et parsemer de fromage rapé.
	\step \textbf{Attention:} bien veiller à recouvrir les pâtes de sauce afin qu'elles soient cuites entièrement.
	\step Préchauffer le four à \unit[200]{ºC} et cuire ± \unit[20]{minutes}.
}

\hint{Les préparer à l'avance les rendront encore plus savoureuses!}

\end{recipe}
