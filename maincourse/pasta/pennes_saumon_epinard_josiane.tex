\begin{recipe}[
 source = Josiane Cerfontaine
]{Pennes saumon épinards}
\ingredients{
	4 & Filets de saumon atlantique \\
	1 & Pot de tomates marinées \\
	\unit[300]{g} & Épinards en branches au Philadelphia \\
	\unit[500]{g} & Pennes fraîches ou \unit[160]{g} de pennes sèches \\
	\unit[1]{botte} & Aneth fraîche \\
	\unit[2]{c à s} & Huile d'olive vièrge \\
}

	\index{Source!Josiane Cerfontaine}
	\index{Plat!Pâtes!Josiane}
	\index{Ingrédients!Poissons!Saumon}
	\index{Ingrédients!Légumes!Épinard}


\preparation {
	\step Préparer les pâtes selon les instructions indiquées sur l'emballage.
	\step Couper le saumon, les épinards et les tomates in morceaux.
	\step Faire revenir le saumon dégelé au wok, avec un peu d'huile.
	\step Hacher l'aneth.
	\step Mélanger le wok aux pâtes.
	\step Ajouter les tomates marinées puis l'aneth hachée.
}
\end{recipe}
